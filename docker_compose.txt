#   Build: docker-compose build

version: '3.8'

services:
  # Main application service
  serendipity-finder:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: serendipity-finder
    volumes:
      - ./output:/app/output
      - ./data:/app/data
    environment:
      - PYTHONUNBUFFERED=1
      - ANALYSIS_MODE=demo
    command: python serendipity_finder.py
    
  # Web interface service
  web-interface:
    image: nginx:alpine
    container_name: serendipity-web
    ports:
      - "8080:80"
    volumes:
      - ./serendipity_visualization.html:/usr/share/nginx/html/index.html:ro
      - ./assets:/usr/share/nginx/html/assets:ro
    depends_on:
      - serendipity-finder
      
  # Optional: Jupyter notebook for interactive analysis
  jupyter:
    image: jupyter/scipy-notebook:latest
    container_name: serendipity-jupyter
    ports:
      - "8888:8888"
    volumes:
      - .:/home/jovyan/work
    environment:
      - JUPYTER_ENABLE_LAB=yes
    command: start-notebook.sh --NotebookApp.token='' --NotebookApp.password=''

volumes:
  output:
  data:

networks:
  default:
    name: serendipity-network
